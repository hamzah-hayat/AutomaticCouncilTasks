# Convert county via court chaplain council task
# When choosing a new County, go in order of the following:
# 1. Liege Capital
# 2. Liege realm
# 3. Direct Vassals Capitals
# 4. Sub-vassals Capitals
# 5. Any remaining counties.

# When choosing a county, order based on county opinion (lowest opinion are picked first)

act_chaplain_automatic_task_conversion = {
	scope:councillor_liege = {
		# 1. Liege Capital
		if = {
			limit = {
				capital_county = { act_is_potential_county_for_conversion = yes }
			}
			capital_county = {
				save_scope_as = target_county
			}
		}

		# 2. Liege realm
		if = {
			limit = { NOT = { exists = scope:target_county } }
			ordered_directly_owned_province = {
				limit = {
					county = { act_is_potential_county_for_conversion = yes }
				}
				order_by = county_opinion
				position = 0

				county = {
					save_scope_as = target_county
				}
			}
		}

		# 3. Direct Vassals Capitals
		if = {
			limit = {NOT = { exists = scope:target_county } }
			every_vassal = {
				limit = {
					exists = capital_county
					capital_county = { act_is_potential_county_for_conversion = yes }
					highest_held_title_tier = tier_kingdom
				}
				alternative_limit = {
					exists = capital_county
					capital_county = { act_is_potential_county_for_conversion = yes }
					highest_held_title_tier = tier_duchy
				}
				alternative_limit = {
					exists = capital_county
					capital_county = { act_is_potential_county_for_conversion = yes }
					highest_held_title_tier = tier_county
				}
				capital_county = {
					add_to_temporary_list = possible_counties
				}
			}
			ordered_in_list = {
				list = possible_counties
				order_by = county_opinion
				position = 0
				save_scope_as = target_county
			}
		}

		# 4. Sub-vassal Capitals
		if = {
			limit = { NOT = { exists = scope:target_county } }
			every_vassal_or_below = {
				alternative_limit = {
					exists = capital_county
					capital_county = { act_is_potential_county_for_conversion = yes }
					highest_held_title_tier = tier_duchy
				}
				alternative_limit = {
					exists = capital_county
					capital_county = { act_is_potential_county_for_conversion = yes }
					highest_held_title_tier = tier_county
				}
				capital_county = {
					add_to_temporary_list = possible_counties
				}
			}
			ordered_in_list = {
				list = possible_counties
				order_by = county_opinion
				position = 0
				save_scope_as = target_county
			}
		}

		# 5. Any remaining counties.
		if = {
			limit = { NOT = { exists = scope:target_county } }
			ordered_sub_realm_county = {
				order_by = county_opinion
				position = 0
				save_scope_as = target_county
			}
		}

		if = {
			limit = { exists = scope:target_county }
			scope:councillor = {
				# have to use an event because otherwise setting new convert task doesn't work?
				trigger_event = act_chaplain_automation.0001
			}
		}
	}
}

act_chaplain_automatic_task_conversion_start = {
	set_council_task = {
		task_type = task_conversion
		target = scope:target_county.title_province
	}
}
